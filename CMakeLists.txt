    GENERATE_TARGET_HEADER "${ROOT}/src/fs/fs_info.hpp"

    CPP_WARNINGS ALL SANE FATAL

    COMPILE_DEFINITIONS ${fs_COMPILE_DEFINITIONS}

        LIB shl 0.10.0 "${ROOT}/ext/shl" INCLUDE LINK GIT_SUBMODULE
